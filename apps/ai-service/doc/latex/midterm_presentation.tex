% Slide 6: Database Internal
\begin{frame}[fragile]{数据库内核：基于递归 CTE 的图谱遍历}
    \textbf{痛点}: 图谱多级跳跃 (Multi-hop) 的传统 IO 瓶颈。\\
    \textbf{解法}: 将层级递归逻辑下沉至 PostgreSQL 内核执行。
    \begin{lstlisting}[language=SQL]
    WITH RECURSIVE graph_traversal AS (
        SELECT id, 0 as hop_level, ARRAY[id] as path FROM knowledge_nodes
        WHERE id IN (SEED_IDS)
        UNION ALL
        SELECT n.id, gt.hop_level + 1, gt.path || n.id FROM graph_traversal gt
        JOIN knowledge_edges e ON gt.id = e.source_id OR gt.id = e.target_id
        JOIN knowledge_nodes n ON n.id = CASE WHEN gt.id = e.source_id THEN e.target_id ELSE e.source_id END
        WHERE gt.hop_level < :max_hops AND NOT n.id = ANY(gt.path)
    )
    SELECT DISTINCT ON (id) * FROM graph_traversal ORDER BY id, hop_level ASC;
    \end{lstlisting}
    \vspace{0.1cm}
    \textbf{性能增益}: 对比传统应用层递归，响应速度实现 \textbf{10 倍以上} 提升。
\end{frame}
